 87    def make_index_references(self, references: list[Reference]) -> list[Reference]:
 88        """Return a sorted list of References; no combining or simplifying."""
 89        index = []
 90        collation = self.collate(
 91            sorted([ref for ref in references if ref and not ref.is_book()])
 92        )
 93        for _, book_collation in collation:
 94            for _, reference_list in book_collation:
 95                sorted_ref = self.sort_references(reference_list)
 96                index.append(sorted_ref)
 97        return index

193    def make_hotspot_tuples(
194        self,
195        references: list[Reference],
196        max_chapters: int = 7,
197        min_references: int = 2,
198    ) -> list[tuple[Reference, int]]:
199        """
200        Find the most referenced chapters in a set of references.
201
202        Args:
203            max_chapters: The maximum number of chapter hotspots to return.
204            min_references: The minimal references per chapter that qualifies as a hotspot.
205
206        Return:
207            a list of tuples of (chapter reference, count) in descending frequency
208        """
209        if not references:
210            return []
211        totals = dict()
212        for _ in self.sort_references(references).ranges:
213            for tuple in [
214                (_.start.library, _.start.book, _.start.chapter),
215                (_.end.library, _.end.book, _.end.chapter),
216            ]:
217                if tuple in totals:
218                    totals[tuple] += 1
219                else:
220                    totals[tuple] = 1
221        hotspots = [
222            (chapter_reference(*tuple), int(total / 2))
223            for tuple, total in totals.items()
224            if total / 2 >= min_references
225        ]
226        hotspots_desc = sorted(hotspots, key=lambda item: item[1], reverse=True)
227        return hotspots_desc[:max_chapters]

272    def sort_references(self, references: list[Reference]) -> Reference:
273        """For a list of references, make a single reference containing their sorted ranges."""
274
275        ranges = []
276        for ref in references:
277            for rng in ref.ranges:
278                ranges.append(rng)
279        return reference(*sorted(ranges))
280
281    def merge_references(self, references: list[Reference]) -> Reference:
282        """For a list of references, merge their ranges into a new reference.
283
284        Merging means combining ranges that overlap.
285        """
286        ranges = []
287        for ref in references:
288            ranges.extend(ref.ranges)
289        new_ref = reference(*merge(ranges))
290        return new_ref
291
292    def combine_references(self, references: list[Reference]) -> Reference:
293        """For a list of references, combine their ranges into a new reference.
294
295        Combining means sorting ranges, merging overlaps, and joining adjacent
296        records.
297        """
298        ranges = []
299        for ref in references:
300            ranges.extend(ref.ranges)
301        new_ref = reference(*combine(ranges))
302        return new_ref

317    def collate_chapter_references(
318        self, references: list[Reference]
319    ) -> list[tuple[Library, list[tuple[Book, list[tuple[Number, list[Reference]]]]]]]:
320        library_list = list()
321        for library_id, books_dict in self.collate_by_chapter(references).items():
322            book_list = list()
323            for book_id, chapters_dict in books_dict.items():
324                chapter_list = list()
325                for chapter_id, references in chapters_dict.items():
326                    chapter_list.append(tuple([chapter_id, references]))
327                book_list.append(tuple([self.books[library_id, book_id], chapter_list]))
328            library_list.append(tuple([self.libraries[library_id], book_list]))
329        return library_list

331    def collate_by_chapter(
332        self, references: list[Reference]
333    ) -> dict[Number, dict[Number, dict[Number, list[Reference]]]]:
334        """
335        A collation groups single-book references by library, book, and chapter
336        IDs. Multi-book references are ignored.
337
338        TODO: Refactor code duplication with collate_by_book.
339        """
340        collation = dict()
341        for ref in [_ for _ in references if _.count_books() == 1]:
342            v1 = ref.ranges[0].start
343            if v1.library not in collation:
344                collation[v1.library] = dict()
345            if v1.book not in collation[v1.library]:
346                collation[v1.library][v1.book] = dict()
347            if v1.chapter not in collation[v1.library][v1.book]:
348                collation[v1.library][v1.book][v1.chapter] = list()
349            collation[v1.library][v1.book][v1.chapter].append(ref)
350        return collation

442    def bcv(
443        self,
444        alias: str,
445        c: Number | None = None,
446        v: Number | None = None,
447        v_end: Number | None = None,
448    ) -> Reference:
449        """Construct a reference from a book alias and chapter/verse numbers.
450
451        Omitting optional arguments will construct a reference to a whole book
452        or chapter.
453
454        Args:
455            v_end: constructs a verse range from `v` to `v_end`.
456
457        Raises:
458            ValueError: If the book_alias is missing, or the number values are
459                out of range.
460        """
461        if alias not in self.book_aliases:
462            raise ValueError(f'Book alias "{alias}" not found.')
463        library_id, book_id = self.book_aliases[alias]
464        ta = TypeAdapter(Number)
465        if c is None:
466            return book_reference(library_id, book_id)
467        if v is None:
468            return chapter_reference(library_id, book_id, ta.validate_python(c))
469
470        if not v_end:
471            return verse_reference(
472                library_id,
473                book_id,
474                ta.validate_python(c),
475                ta.validate_python(v),
476            )
477        else:
478            return reference(
479                range(
480                    verse(
481                        library_id,
482                        book_id,
483                        ta.validate_python(c),
484                        ta.validate_python(v),
485                    ),
486                    verse(
487                        library_id,
488                        book_id,
489                        ta.validate_python(c),
490                        ta.validate_python(v_end),
491                    ),
492                )
493            )

495    def bcr(
496        self, alias: str, c: Number, v_ranges: list[Number | tuple[Number, Number]]
497    ) -> Reference:
498        if alias not in self.book_aliases:
499            raise ValueError(f'Book alias "{alias}" not found.')
500        library_id, book_id = self.book_aliases[alias]
501        ranges = []
502        for val in v_ranges:
503            if isinstance(val, tuple):
504                v_start, v_end = val
505                ranges.append(
506                    range(
507                        verse(library_id, book_id, c, v_start),
508                        verse(library_id, book_id, c, v_end),
509                    )
510                )
511            elif isinstance(val, int):
512                v_start = v_end = val
513                ranges.append(
514                    range(
515                        verse(library_id, book_id, c, v_start),
516                        verse(library_id, book_id, c, v_end),
517                    )
518                )
519        return reference(*ranges)

622    def template(self, reference: Reference | None, pattern: str | None = None) -> str:
623        """
624        Substitute formatting values in a string:
625
626            * `{LINK}` -> "1%20Cor%202:3-4"
627                    - like ESC_ABBREV_NAME, but with English numbering in any language.
628                    - useful for BibleGateway and presumably other sites.
629            * `{NAME}` -> "1 Corinthians 2:34"
630            * `{BOOK}` -> "1 Corinthians"
631            * `{NUMBERS}` -> "2:34"
632            * `{ABBREV_NAME}` -> "1 Cor 2:34"
633            * `{ABBREV_BOOK}` -> "1 Cor"
634            * `{ABBREV_NUMBERS}` -> "2:34"
635            * `{ESC_NAME}` -> "1%20Corinthians%202%3A3-4"
636            * `{ESC_BOOK}` -> "1%20Corinthians"
637            * `{ESC_NUMBERS}` -> "2%3A3-4"
638            * `{ESC_ABBREV_NAME}` -> "1%20Cor%202%3A3-4"
639            * `{ESC_ABBREV_BOOK}` -> "1%20Cor"
640            * `{ESC_ABBREV_NUMBERS}` -> "2%3A3-4"
641            * `{PARAM_NAME}`-> "1cor+2.3-4"
642            * `{PARAM_BOOK}` -> "1cor"
643            * `{PARAM_NUMBERS}` -> "2.3-4"
644
645        For efficiency, we calculate only the values required by the template
646        string.
647        """
648        if reference is None:
649            return ""
650
651        out = pattern if pattern else "{ABBREV_NAME}"
652
653        regexp = re.compile(r"\{[A-Z_]+\}")
654        matches = regexp.findall(out)
655        numbers = self.numbers(reference)
656        for _ in matches:
657            if _ == "{LINK}":
658                out = out.replace("{LINK}", url_escape(self.link(reference)))
659            elif _ == "{NAME}":
660                out = out.replace("{NAME}", self.name(reference))
661            elif _ == "{BOOK}":
662                out = out.replace("{BOOK}", self.book(reference))
663            elif _ == "{NUMBERS}":

194    def make_hotspot_tuples(
195        self,
196        references: list[Reference],
197        max_chapters: int = 7,
198        min_references: int = 2,
199    ) -> list[tuple[Reference, int]]:
200        """
201        Find the most referenced chapters in a set of references.
202
203        Args:
204            max_chapters: The maximum number of chapter hotspots to return.
205            min_references: The minimal references per chapter that qualifies as a hotspot.
206
207        Return:
208            a list of tuples of (chapter reference, count) in descending frequency
209        """
210        if not references:
211            return []
212        totals = dict()
213        for _ in self.sort_references(references).ranges:
214            for tuple in [
215                (_.start.library, _.start.book, _.start.chapter),
216                (_.end.library, _.end.book, _.end.chapter),
217            ]:
218                if tuple in totals:
219                    totals[tuple] += 1
220                else:
221                    totals[tuple] = 1
222        hotspots = [
223            (chapter_reference(*tuple), int(total / 2))
224            for tuple, total in totals.items()
225            if total / 2 >= min_references
226        ]
227        hotspots_desc = sorted(hotspots, key=lambda item: item[1], reverse=True)
228        return hotspots_desc[:max_chapters]

623    def template(self, reference: Reference | None, pattern: str | None = None) -> str:
624        """
625        Substitute formatting values in a string:
626
627            * `{LINK}` -> "1%20Cor%202:3-4"
628                    - like ESC_ABBREV_NAME, but with English numbering in any language.
629                    - useful for BibleGateway and presumably other sites.
630            * `{NAME}` -> "1 Corinthians 2:34"
631            * `{BOOK}` -> "1 Corinthians"
632            * `{NUMBERS}` -> "2:34"
633            * `{ABBREV_NAME}` -> "1 Cor 2:34"
634            * `{ABBREV_BOOK}` -> "1 Cor"
635            * `{ABBREV_NUMBERS}` -> "2:34"
636            * `{ESC_NAME}` -> "1%20Corinthians%202%3A3-4"
637            * `{ESC_BOOK}` -> "1%20Corinthians"
638            * `{ESC_NUMBERS}` -> "2%3A3-4"
639            * `{ESC_ABBREV_NAME}` -> "1%20Cor%202%3A3-4"
640            * `{ESC_ABBREV_BOOK}` -> "1%20Cor"
641            * `{ESC_ABBREV_NUMBERS}` -> "2%3A3-4"
642            * `{PARAM_NAME}`-> "1cor+2.3-4"
643            * `{PARAM_BOOK}` -> "1cor"
644            * `{PARAM_NUMBERS}` -> "2.3-4"
645
646        For efficiency, we calculate only the values required by the template
647        string.
648        """
649        if reference is None:
650            return ""
651
652        out = pattern if pattern else "{ABBREV_NAME}"
653
654        regexp = re.compile(r"\{[A-Z_]+\}")
655        matches = regexp.findall(out)
656        numbers = self.numbers(reference)
657        for _ in matches:
658            if _ == "{LINK}":
659                out = out.replace("{LINK}", url_escape(self.link(reference)))
660            elif _ == "{NAME}":
661                out = out.replace("{NAME}", self.name(reference))
662            elif _ == "{BOOK}":
663                out = out.replace("{BOOK}", self.book(reference))
664            elif _ == "{NUMBERS}":
665                out = out.replace("{NUMBERS}", numbers)
666            elif _ == "{ASCII_NUMBERS}":
667                out = out.replace("{ASCII_NUMBERS}", numbers.replace("", "-"))
668            elif _ == "{ABBREV_NAME}":
669                out = out.replace("{ABBREV_NAME}", self.abbrev_name(reference))

623    def template(self, reference: Reference | None, pattern: str | None = None) -> str:
624        """

626

628                    - like ESC_ABBREV_NAME, but with English numbering in any language.
629                    - useful for BibleGateway and presumably other sites.
630            * `{NAME}` -> "1 Corinthians 2:34"
631            * `{BOOK}` -> "1 Corinthians"
632            * `{NUMBERS}` -> "2:34"
633            * `{ABBREV_NAME}` -> "1 Cor 2:34"
634            * `{ABBREV_BOOK}` -> "1 Cor"
635            * `{ABBREV_NUMBERS}` -> "2:34"
636            * `{ESC_NAME}` -> "1%20Corinthians%202%3A3-4"
637            * `{ESC_BOOK}` -> "1%20Corinthians"
638            * `{ESC_NUMBERS}` -> "2%3A3-4"
639            * `{ESC_ABBREV_NAME}` -> "1%20Cor%202%3A3-4"
640            * `{ESC_ABBREV_BOOK}` -> "1%20Cor"
641            * `{ESC_ABBREV_NUMBERS}` -> "2%3A3-4"
642            * `{PARAM_NAME}`-> "1cor+2.3-4"
643            * `{PARAM_BOOK}` -> "1cor"
644            * `{PARAM_NUMBERS}` -> "2.3-4"
645
646        For efficiency, we calculate only the values required by the template
647        string.
648        """
649        if reference is None:
650            return ""
651
652        out = pattern if pattern else "{ABBREV_NAME}"
653
654        regexp = re.compile(r"\{[A-Z_]+\}")
655        matches = regexp.findall(out)
656        numbers = self.numbers(reference)

660            elif _ == "{NAME}":

662            elif _ == "{BOOK}":

664            elif _ == "{NUMBERS}":

666            elif _ == "{ASCII_NUMBERS}":
667                out = out.replace("{ASCII_NUMBERS}", numbers.replace("", "-"))
668            elif _ == "{ABBREV_NAME}":
669                out = out.replace("{ABBREV_NAME}", self.abbrev_name(reference))
670            elif _ == "{ABBREV_BOOK}":
671                out = out.replace("{ABBREV_BOOK}", self.abbrev_name(reference))
672            elif _ == "{ESC_NAME}":
673                out = out.replace("{ESC_NAME}", url_escape(self.name(reference)))
674            elif _ == "{ESC_BOOK}":
675                out = out.replace("{ESC_BOOK}", url_escape(self.book(reference)))
676            elif _ == "{ESC_NUMBERS}":

678            elif _ == "{ESC_ASCII_NUMBERS}":
679                out = out.replace(
680                    "{ESC_ASCII_NUMBERS}",
681                    url_escape(numbers.replace("", "-")),
682                )
683            elif _ == "{ESC_ABBREV_NAME}":
684                out = out.replace(
